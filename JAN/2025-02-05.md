import pandas as pd

data = {
    '이름': ['홍길동', '김철수', '박영희', '이순신', '강감찬', '신사임당'],

    '급여': [5000, 5500, 4800, 5100, 6000, 6200]
}

df = pd.DataFrame(data)
df1 = df.groupby('부서').filter(lambda x: x['급여'].mean() >= 5000)
df2 = df.groupby('부서').apply(lambda x: x['급여'].mean() >= 5000) # 이것은 '적용'이기에 출력값이 불리안으로 나옴
